cmake_minimum_required(VERSION 3.20)
project(iidm-bridge-cpp VERSION 1.0.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

option(IIDM_BRIDGE_ENABLE_GRAALVM "Build the GraalVM backend" ON)
option(IIDM_BRIDGE_ENABLE_JNI     "Build the JNI backend" ON)
option(IIDM_BRIDGE_BUILD_TESTS    "Build unit tests" ON)
option(IIDM_BRIDGE_BUILD_EXAMPLES "Build examples" ON)
option(IIDM_BRIDGE_BUILD_JAVA     "Build the Java fat-JAR via Maven" ON)
option(IIDM_BRIDGE_BUILD_NATIVE   "Build the GraalVM native shared image via native-image" ON)

# ── JNI ──────────────────────────────────────────────────────────────────────
if(IIDM_BRIDGE_ENABLE_JNI)
    find_package(JNI REQUIRED)
endif()

# ── GraalVM ──────────────────────────────────────────────────────────────────
if(IIDM_BRIDGE_ENABLE_GRAALVM)
    if(IIDM_BRIDGE_BUILD_NATIVE)
        # graal_isolate.h is generated by the iidmbridge_native target below;
        # point GRAALVM_INCLUDE_DIRS at the output directory now so the
        # compile command is correct once the build runs.
        set(GRAALVM_INCLUDE_DIRS "${CMAKE_BINARY_DIR}/graalvm-native")
    elseif(NOT GRAALVM_INCLUDE_DIRS)
        # No pre-set path: search via FindGraalVM.cmake or GRAALVM_HOME.
        list(APPEND CMAKE_MODULE_PATH "${CMAKE_SOURCE_DIR}/cmake")
        find_package(GraalVM)
        if(NOT GraalVM_FOUND)
            message(WARNING "GraalVM not found; GraalVM backend will compile without SDK headers. "
                            "Set GRAALVM_HOME or -DGRAALVM_INCLUDE_DIRS to enable full GraalVM support.")
        endif()
    endif()
endif()

# ── Library sources ──────────────────────────────────────────────────────────
set(IIDM_BRIDGE_SOURCES
    src/Network.cpp
    src/NetworkFactory.cpp
    src/BackendSelector.cpp
    src/ActivePowerControl.cpp
    src/Generator.cpp
    src/Load.cpp
    src/Terminal.cpp
    src/Bus.cpp
    src/Substation.cpp
    src/VoltageLevel.cpp
    src/Line.cpp
    src/TwoWindingsTransformer.cpp
    src/ThreeWindingsTransformer.cpp
    src/HvdcLine.cpp
    src/DanglingLine.cpp
    src/ShuntCompensator.cpp
    src/StaticVarCompensator.cpp

    src/LccConverterStation.cpp

if(IIDM_BRIDGE_ENABLE_GRAALVM)
    list(APPEND IIDM_BRIDGE_SOURCES
        src/graalvm/GraalVMBackend.cpp)
endif()

if(IIDM_BRIDGE_ENABLE_JNI)
    list(APPEND IIDM_BRIDGE_SOURCES
        src/jni/JNIBackend.cpp)
endif()

# On Windows/MSVC, export all non-inline symbols so that the linker generates
# iidmbridge.lib (the import library).  On Linux/macOS this is a no-op.
set(CMAKE_WINDOWS_EXPORT_ALL_SYMBOLS ON)

add_library(iidmbridge SHARED ${IIDM_BRIDGE_SOURCES})

target_include_directories(iidmbridge
    PUBLIC
        $<BUILD_INTERFACE:${CMAKE_SOURCE_DIR}/include>
        $<INSTALL_INTERFACE:include>
    PRIVATE
        $<$<BOOL:${IIDM_BRIDGE_ENABLE_JNI}>:${JNI_INCLUDE_DIRS}>
        ${CMAKE_SOURCE_DIR}/src
        $<$<BOOL:${IIDM_BRIDGE_ENABLE_GRAALVM}>:${GRAALVM_INCLUDE_DIRS}>
)

target_link_libraries(iidmbridge
    PRIVATE
        $<$<BOOL:${IIDM_BRIDGE_ENABLE_JNI}>:${JNI_LIBRARIES}>
        ${CMAKE_DL_LIBS}
)

target_compile_definitions(iidmbridge PRIVATE
    $<$<BOOL:${IIDM_BRIDGE_ENABLE_GRAALVM}>:IIDM_BRIDGE_GRAALVM_ENABLED>
    $<$<BOOL:${IIDM_BRIDGE_ENABLE_JNI}>:IIDM_BRIDGE_JNI_ENABLED>
)

# ── Installation & CMake config ───────────────────────────────────────────────
include(GNUInstallDirs)
include(CMakePackageConfigHelpers)

install(TARGETS iidmbridge EXPORT IidmBridgeTargets
    LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
    ARCHIVE DESTINATION ${CMAKE_INSTALL_LIBDIR}
    RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
    INCLUDES DESTINATION ${CMAKE_INSTALL_INCLUDEDIR}
)
install(DIRECTORY include/ DESTINATION ${CMAKE_INSTALL_INCLUDEDIR})
install(EXPORT IidmBridgeTargets
    FILE IidmBridgeTargets.cmake
    NAMESPACE IidmBridge::
    DESTINATION ${CMAKE_INSTALL_LIBDIR}/cmake/IidmBridge
)

    "${CMAKE_CURRENT_BINARY_DIR}/IidmBridgeConfig.cmake"
    INSTALL_DESTINATION ${CMAKE_INSTALL_LIBDIR}/cmake/IidmBridge

write_basic_package_version_file(
    "${CMAKE_CURRENT_BINARY_DIR}/IidmBridgeConfigVersion.cmake"
    VERSION ${PROJECT_VERSION}
    COMPATIBILITY SameMajorVersion
)

install(FILES
    "${CMAKE_CURRENT_BINARY_DIR}/IidmBridgeConfig.cmake"
    "${CMAKE_CURRENT_BINARY_DIR}/IidmBridgeConfigVersion.cmake"
    DESTINATION ${CMAKE_INSTALL_LIBDIR}/cmake/IidmBridge
)

# ── Tests ─────────────────────────────────────────────────────────────────────
if(IIDM_BRIDGE_BUILD_TESTS)
    enable_testing()
    find_package(GTest QUIET)
    if(NOT GTest_FOUND)
        message(STATUS "GTest not found locally; fetching from GitHub...")
        include(FetchContent)
        FetchContent_Declare(
            googletest
            GIT_REPOSITORY https://github.com/google/googletest.git
            GIT_TAG        v1.14.0
            GIT_SHALLOW    TRUE
        )
        # Keep gtest from overriding the CRT on Windows (important for MSVC)
        set(gtest_force_shared_crt ON CACHE BOOL "" FORCE)
        set(BUILD_GMOCK OFF CACHE BOOL "" FORCE)
        set(INSTALL_GTEST OFF CACHE BOOL "" FORCE)
        FetchContent_MakeAvailable(googletest)
        # Alias to the target names that find_package(GTest) normally produces
        add_library(GTest::GTest ALIAS gtest)
        add_library(GTest::Main  ALIAS gtest_main)
    endif()
    add_subdirectory(test)
endif()

# ── Java fat-JAR and GraalVM native image ────────────────────────────────────
#
# IIDM_BRIDGE_BUILD_JAVA=ON   – runs 'mvn package -DskipTests' and exposes
#                               an 'iidmbridge_jar' target.
# IIDM_BRIDGE_BUILD_NATIVE=ON – builds libpowsybl-iidm-native from the fat-JAR
#                               via 'native-image --shared'; also produces
#                               graal_isolate.h (required when GRAALVM=ON).
#
# Typical "build everything from source" invocation:
#   cmake -S . -B build \
#     -DIIDM_BRIDGE_ENABLE_GRAALVM=ON \
#     -DIIDM_BRIDGE_ENABLE_JNI=ON \
#     -DIIDM_BRIDGE_BUILD_JAVA=ON \
#     -DIIDM_BRIDGE_BUILD_NATIVE=ON
#   cmake --build build --parallel
#
if(IIDM_BRIDGE_BUILD_JAVA OR IIDM_BRIDGE_BUILD_NATIVE)
    messagge("We will build fat jar.")
    find_program(MAVEN_EXECUTABLE mvn)
    if(NOT MAVEN_EXECUTABLE)
        message(FATAL_ERROR
            "Maven (mvn) not found on PATH. "
            "It is required when IIDM_BRIDGE_BUILD_JAVA or IIDM_BRIDGE_BUILD_NATIVE is ON.")
    endif()

    # Override IIDM_BRIDGE_JAVA_VERSION if the <version> in java/pom.xml changes.
    set(IIDM_BRIDGE_JAVA_VERSION "1.0.0-SNAPSHOT" CACHE STRING
        "Java module version; must match the <version> tag in java/pom.xml")
    set(_FAT_JAR
        "${CMAKE_SOURCE_DIR}/java/target/iidm-bridge-java-${IIDM_BRIDGE_JAVA_VERSION}-fat.jar")

    add_custom_command(
        OUTPUT  "${_FAT_JAR}"
        COMMAND "${MAVEN_EXECUTABLE}" --batch-mode --no-transfer-progress package -DskipTests
        WORKING_DIRECTORY "${CMAKE_SOURCE_DIR}/java"
        COMMENT "Building Java fat-JAR"
        VERBATIM
    )
    add_custom_target(iidmbridge_jar DEPENDS "${_FAT_JAR}")
endif()

if(IIDM_BRIDGE_BUILD_NATIVE)
    find_program(NATIVE_IMAGE_EXECUTABLE native-image)
    if(NOT NATIVE_IMAGE_EXECUTABLE)
        message(FATAL_ERROR
            "native-image not found on PATH. "
            "Install GraalVM CE and run 'gu install native-image'.")
    endif()

    set(_NATIVE_DIR "${CMAKE_BINARY_DIR}/graalvm-native")
    # Create the output directory at configure time so it exists as the
    # WORKING_DIRECTORY when the custom command runs at build time.
    file(MAKE_DIRECTORY "${_NATIVE_DIR}")

    if(WIN32)
        set(_NATIVE_LIB "${_NATIVE_DIR}/libpowsybl-iidm-native.dll")
    elseif(APPLE)
        set(_NATIVE_LIB "${_NATIVE_DIR}/libpowsybl-iidm-native.dylib")
    else()
        set(_NATIVE_LIB "${_NATIVE_DIR}/libpowsybl-iidm-native.so")
    endif()
    set(_NATIVE_H "${_NATIVE_DIR}/graal_isolate.h")

    add_custom_command(
        OUTPUT  "${_NATIVE_LIB}" "${_NATIVE_H}"
        COMMAND "${NATIVE_IMAGE_EXECUTABLE}"
                --shared
                -cp "${_FAT_JAR}"
                -H:+ReportExceptionStackTraces
                -march=compatibility
                --no-fallback
                --gc=serial
                -o "${_NATIVE_DIR}/libpowsybl-iidm-native"
                --allow-incomplete-classpath
                "--add-exports=org.graalvm.nativeimage.builder/com.oracle.svm.core.c=ALL-UNNAMED"
        DEPENDS "${_FAT_JAR}"
        WORKING_DIRECTORY "${_NATIVE_DIR}"
        COMMENT "Building GraalVM native image (libpowsybl-iidm-native)"
        VERBATIM
    )
    add_custom_target(iidmbridge_native DEPENDS "${_NATIVE_LIB}" "${_NATIVE_H}")

    # Install the native shared library so cmake --install bundles it.
    install(FILES "${_NATIVE_LIB}" DESTINATION ${CMAKE_INSTALL_LIBDIR})

    # iidmbridge must not be compiled before graal_isolate.h is available.
    if(IIDM_BRIDGE_ENABLE_GRAALVM)
        add_dependencies(iidmbridge iidmbridge_native)
    endif()
endif()
